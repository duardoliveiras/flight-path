735double distanceEarth(double latOrigin, double longOrigin, double latDest,
736 double longDest)
737{
738 latOrigin = toRadians(latOrigin);
739 longOrigin = toRadians(longOrigin);
740 latDest = toRadians(latDest);
741 longDest = toRadians(longDest);
742
743 double dlong = longDest - longOrigin;
744 double dlat = latDest - latOrigin;
745
746 double ans = pow(sin(dlat / 2), 2) +
747 cos(latOrigin) * cos(latDest) * pow(sin(dlong / 2), 2);
748
749 ans = 2 * asin(sqrt(ans));
750
751 double R = 6371;
752
753 ans = ans * R;
754
755 return ans;
756}

964vector<vector<Flight>> bfsPath(Vertex<Airport> *v, string &tgt,
965 vector<string> &airplanes)
966{
967 vector<vector<Flight>> paths;
968 unordered_map<string, vector<Flight>> prev;
969 unordered_map<string, int> dist;
970 queue<Vertex<Airport> *> q;
971
972 q.push(v);
973 v->setVisited(true);
974 dist[v->getInfo().getCode()] = 0;
975
976 while (!q.empty())
977 {
978 auto vertex = q.front();
979 q.pop();
980 auto adjs = vertex->getAdj();
981
982 for (auto &e : adjs)
983 {
984 auto w = e.getDest();
985 if (!w->isVisited() && (find(airplanes.begin(), airplanes.end(),
986 e.getRoute()) != airplanes.end() ||
987 airplanes.size() == 0))
988 {
989 q.push(w);
990 w->setVisited(true);
991 prev[w->getInfo().getCode()].push_back(
992 {vertex->getInfo().getCode(), e.getRoute()});
993 dist[w->getInfo().getCode()] = dist[vertex->getInfo().getCode()] + 1;
994 }
995 else if (dist[w->getInfo().getCode()] ==
996 dist[vertex->getInfo().getCode()] + 1 &&
997 (find(airplanes.begin(), airplanes.end(), e.getRoute()) !=
998 airplanes.end() ||
999 airplanes.size() == 0))
1000 {
1001 prev[w->getInfo().getCode()].push_back(
1002 {vertex->getInfo().getCode(), e.getRoute()});
1003 }
1004 }
1005 }
1006
1007 vector<Flight> path;
1008 getPath(tgt, path, prev, paths, v->getInfo().getCode(), "");
1009
1010 return paths;
